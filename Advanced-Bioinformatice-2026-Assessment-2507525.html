# we now need a quadratic model 
# reaction distance is proportional to speed and braking distance is proportional to speed squared
# and total stopping distance = reaction distance + braking distance
newfit <- lm(dist ~ speed + I(speed^2), data = cars)

The given reaction time is 0.91329 in ft/mph. To convert this into seconds for ‘reaction time in seconds’ we must convert from mph to ft/s. To do this divide by 1.467 as 1 mph = 1.467 ft/s.

0.91329 / 1.467 = an average reaction time of 0.62 seconds

This seems to be a reasonable average reaction time for braking. Average reaction time for braking is usually estimated between 0.5-1.5 seconds. However, the standard error is large, indicating that the result is not reliable, and neither speed or speed^2 coefficients are statistically significant.

rownames(col_data) <- colnames(counts)
# constructing DESeq from count matrix and sample desc 
# design = ~ condition specifies that condition is the variable of interest

Task 3.11

Perform rlog and VST transformation on the data.

# rlog transformation to stabilise variance across genes
# blind = TRUE means DESeq2 will ignore experimental conditions when transforming data 
# assay() extracts the transformed matrix (in this case from the rlog object)
rld <- rlog(dds, blind = TRUE)
rld_counts <- assay(rld)
class(rld_counts)
## [1] "matrix" "array"
# vst transformation for similar variance normalisation
vsd <- vst(dds, blind = TRUE)
vsd_counts <- assay(vsd)
class(vsd_counts)
## [1] "matrix" "array"
# Transforming the data reduces noise and improves visualisation/interpretation
